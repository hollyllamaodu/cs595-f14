    return (typeof(getUrlVars()[s]) != "undefined" && getUrlVars()[s] !== null);
  }

  function readParamsURL() {
    // Allow the button destinations, tweet text, banner text, and use of experiment to be
    // set via the query string. This will be helpful if have a banner installed on one
    // page but want to be able to point different forms at it and reference the action the
    // user just took. Functionally, this works by over-writing parameters in the 
    // thanks_banner_params variable (discussed below), so GET parameters will take precedence
    // over parameters specified along with the node embed.

    // GET Parameters:
    // thx_fb_url:  The URL to point Facebook's sharer.php at (share text will be specified
    //              by that page's open graph metadata)
    //
    // thx_tweet:   Text of the tweet to be shared via the Twitter button.
    //
    // thx_sub_txt: Text to appear in the banner below "Thank You" -- may specify HTML.
    //
    // thx_use_exp: Boolean to specify if randomization should be used between new and old banners.

    var readVars = ['fb_url', 'tweet', 'sub_txt', 'use_exp'];
    var urlParamPre = 'thx_';
    for (var i=0; i < readVars.length; i++) {
      if (inQueryString(urlParamPre+readVars[i])) {
        thanks_banner_params[readVars[i]] = decodeURIComponent(getUrlVars()[urlParamPre+readVars[i]]);
      }
    }

  }

  function applyParams() {
    // Customize the banner based on admin-specified parameters or defaults. The parameters can
    // be set in a script tag along with the node embed using a dictionary called
    // `thanks_banner_params` -- note that this should be fired immediately upon load, and thus
    // NOT wrapped in $(document).ready(). Upon finishing this set-up, the `use_exp` boolean is
    // returned to allow the calling function to randomize if necessary.

    // thanks_banner_params keys:
    // fb_url:  The URL to point Facebook's sharer.php at (share text will be specified
    //          by that page's open graph metadata)
    //
    // tweet:   Text of the tweet to be shared via the Twitter button.
    //
    // sub_txt: Text to appear in the banner below "Thank You" -- may specify HTML.
    //
    // use_exp: Boolean to specify if randomization should be used between new and old banners.


    // Set up some defaults. If a share URL hasn't been specified, share either the URL specified
    // by this page's og:url parameter or, failing that, the URL of the page itself. Also set up
    // the base URL's for the Facebook & Twitter sharer utilities and default to using randomization
    // to test the new banner versus the old one.
    var fallback_url = $("meta[property='og:url']").attr('content') || window.location.href.replace(/\?[^#]*/, '').replace('m.whitehouse.gov', 'www.whitehouse.gov');
    var fb_sharer = 'https://facebook.com/sharer.php?u=';
    var tw_sharer = 'https://twitter.com/intent/tweet?source=webclient&text=';
    var use_exp = true;

    // Set up the Facebook button
    if (thanks_banner_params['fb_url']) {
        $('#thanks-share-btn-fb').attr('href', fb_sharer + encodeURIComponent(thanks_banner_params['fb_url']));
    } else {
        $('#thanks-share-btn-fb').attr('href', fb_sharer + encodeURIComponent(fallback_url));
    }

    // Set up the Twitter button
    if (thanks_banner_params['tweet']) {
        $('#thanks-share-btn-tw').attr('href', tw_sharer + encodeURIComponent(thanks_banner_params['tweet']));
    } else {
        $('#thanks-share-btn-tw').attr('href', tw_sharer + encodeURIComponent('Check out the White House website ' + fallback_url));
    }

    // Set up the banner text
    if (thanks_banner_params['sub_txt']) {
        $('#submission-thank-you p.sub-text').html(thanks_banner_params['sub_txt']);
    }

    // Pull in the experiment flag, which could be a string if specified in a GET parameter
    // or accidently set using quotes in the dictionary.
    if (typeof(thanks_banner_params['use_exp']) == 'boolean') {
        use_exp = thanks_banner_params['use_exp'];
    } else if (typeof(thanks_banner_params['use_exp']) == 'string') {
        use_exp = (thanks_banner_params['use_exp'].toLowerCase().trim() === 'true');
    }

    return use_exp;
  }

<div class="grid-14 progress-box-image-teal-2 left image-7"><div class="something third"><img src="/sites/default/files/image/climate/climate_resilient_sandy_1.png"></div></div>
<div class="grid-13 left"><h4 class="teal-2" style="">GOALS</h4><p class="progress-body-teal-2">Provide an assessment of climate change impacts on the United States that translates scientific insights into practical knowledge that can help decision-makers prepare for specific impacts.</p></div>

    <h2 class="section-title">President Obama's Plan Will Lead International Efforts to Address Global Climate Change</h2>

    
    
          <div class="section-body">
          <p class="pre-text">Because climate change spans international borders, the President&#39;s plan will also</p>
<h1 class="big full-width">
	Lead international efforts to address global climate change</h1>
<p class="paragraph">America will continue to take on a leadership role in engaging the world&#39;s major economies to advance key climate priorities and in galvanizing global action through international climate negotiations. The plan will:</p>
<h2 class="green full-width left">
	WORK WITH OTHER COUNTRIES TO TAKE ACTION TO ADDRESS CLIMATE CHANGE</h2>
<div class="grid-27 left">
	<h4 class="green">
		Lead public sector financing toward cleaner energy</h4>
</div>
<div class="grid-28 progress-box-light-green-end left" style="height:150px!important;margin-left: 5px;">
	<div class="progress-check-wrapper">
		<div class="progress-check-light-green progress-check">
			&nbsp;</div>
	</div>
	<h4 class="green" style="">
		PROGRESS:</h4>
	<p class="progress-body-light-green">The President put forth an initiative to end public financing for new coal-fired power plants overseas, except in rare circumstances. Following the lead of the U.S., other nations&mdash;including the U.K., the Netherlands, and the Nordic countries&mdash;have joined the initiative.</p>
</div>
<hr class="thick-light-green full-width left" />
<div class="grid-27 left">
	<h4 class="green">
		Bilat cooperation with major economies</h4>
</div>
<div class="grid-28 progress-box-light-green-end left" style="height:150px!important;margin-left: 5px;">
	<div class="progress-check-wrapper">
		<div class="progress-check-light-green progress-check">
			&nbsp;</div>
	</div>
	<h4 class="green" style="">
		PROGRESS:</h4>
	<p class="progress-body-light-green">President Obama has made climate change a key issue in some of our most important bilateral relations, including China and India. Together, we are making progress around issue areas such as vehicle emissions standards, energy efficiency, and clean energy initiatives.</p>
</div>
<hr class="thick-light-green full-width left" />
<div class="grid-27 left">
	<h4 class="green">
		Expand clean energy use and cut energy waste</h4>
</div>
<div class="grid-28 progress-box-light-green-end left" style="height:150px!important;margin-left: 5px;">
	<div class="progress-check-wrapper">
		<div class="progress-check-light-green progress-check">
			&nbsp;</div>
	</div>
	<h4 class="green" style="">
		PROGRESS:</h4>
	<p class="progress-body-light-green">Facilitating the transition to a global clean energy economy, the U.S. Department of Energy is leading the Clean Energy Ministerial, a high-level global forum that promotes policies and programs aimed at scaling up energy efficiency and clean energy.</p>
</div>
<hr class="thick-light-green full-width left" />
<div class="grid-27 left">
	<h4 class="green">
		COMBAT SHORT-LIVED CLIMATE POLLUTANTS</h4>
</div>
<div class="grid-14 progress-box-light-green left" style="height:235px!important;margin-left:-5px!important;">
	<div class="progress-check-wrapper">
		<div class="progress-check-light-green progress-check">
			&nbsp;</div>
	</div>
	<h4 class="green" style="">
		PROGRESS:</h4>
	<p class="progress-body-light-green">Building on the breakthrough June 2013 agreement on hydrofluorocarbons (HFCs) by President Obama and China&rsquo;s President Xi, G-20 leaders in September 2013 expressed support for using the expertise and institutions of the Montreal Protocol to phase down HFCs.</p>
</div>
<div class="grid-14 progress-box-light-green-end left" style="height:235px!important;">
	<div class="progress-check-wrapper">
		<div class="progress-check-light-green progress-check">
			&nbsp;</div>
	</div>
	<h4 class="green" style="">
		PROGRESS:</h4>
	<p class="progress-body-light-green">The U.S. continues to spearhead the Climate and Clean Air Coalition which has expanded to 88 partners, including 39 countries. The Coalition is implementing ten initiatives to reduce emissions of methane, HFCs, and black carbon.</p>
</div>
<hr class="thick-light-green full-width left" />
<div class="grid-27 left">
	<h4 class="green">
		Reduce emissions from deforestation and forest degradation</h4>
</div>
<div class="grid-28 progress-box-light-green-end left" style="height:150px!important;margin-left: 5px;">
	<div class="progress-check-wrapper">
		<div class="progress-check-light-green progress-check">
			&nbsp;</div>
	</div>
	<h4 class="green" style="">
		PROGRESS:</h4>
	<p class="progress-body-light-green">In November 2013, the U.S., Norway, and the U.K. launched a public-private partnership to support forests in developing countries, with the goal of reducing emissions from deforestation and promoting sustainable agriculture.</p>
</div>
<hr class="thick-light-green full-width left" />
<div class="grid-27 left">
	<h4 class="green">
		NEGOTIATE GLOBAL FREE TRADE IN ENVIRONMENTAL GOODS AND SERVICES</h4>
</div>
<div class="grid-28 progress-box-light-green-end left" style="height:150px!important;margin-left: 5px;">
	<div class="progress-check-wrapper">
		<div class="progress-check-light-green progress-check">
			&nbsp;</div>
	</div>
	<h4 class="green" style="">
		PROGRESS:</h4>
	<p class="progress-body-light-green">In January 2014, a U.S.-led coalition of countries&mdash;representing 86% of global trade in environmental goods&mdash;announced plans to launch talks aimed at eliminating tariffs on a wide range of environmental goods under the World Trade Organization.</p>
</div>
<hr class="thick-light-green full-width left" />
<div class="grid-27 left">
	<h4 class="green">
		ENHANCE MULTILATERAL ENGAGEMENT WITH MAJOR ECONOMIES</h4>
</div>
<div class="grid-28 progress-box-light-green-end left" style="height:150px!important;margin-left: 5px;">
	<div class="progress-check-wrapper">
		<div class="progress-check-light-green progress-check">
			&nbsp;</div>
	</div>
	<h4 class="green" style="">
		PROGRESS:</h4>
	<p class="progress-body-light-green">The United States continues to play an active role in shaping the design of a new global climate agreement due in 2015, including through our chairmanship of the major economies forum on energy and climate.</p>
</div>
<hr class="thick-light-green full-width left" />
<div class="grid-27 left">
	<h4 class="green">
		Mobilize climate finance</h4>
</div>
<div class="grid-28 progress-box-light-green-end left" style="height:150px!important;margin-left: 5px;">
	<div class="progress-check-wrapper">
		<div class="progress-check-light-green progress-check">
			&nbsp;</div>
	</div>
	<h4 class="green" style="">
		PROGRESS:</h4>
	<p class="progress-body-light-green">In April 2014, the U.S., U.K., and Germany announced the Global Innovation Lab for Climate Finance, a public-private platform designed to spur private-sector investment in low-carbon, climate-resilient infrastructure in developing countries.</p>
</div>
<hr class="thick-light-green full-width left" />
<div class="full-width">
	<h2 class="green full-width left">
		Lead efforts to address climate change through international negotiations</h2>
<p class="progress-body-light-green full-width" style="min-width:900px !important;height:35px;">The United States has made historic progress in the international climate negotiations during the past four years.</p>
</div>
